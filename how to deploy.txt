#在linux宿主机上创建一个目录用于保存jar包和Dockerfile（两者需要在同一个目录下）

#使用下列命令，创建一个docker镜像
docker build -t ss-community .

#创建好之后，使用下面的命令，启动容器
docker run --name ss-community -p 80:80 \
 -v /opt/image/ss-community/:/opt/image/ss-community/ \
 -d ss-community


#-d表示在后台运行
#--network host表示与宿主机共享网络
#-v表示使用volume存储镜像文件（容器重启后文件依然存在，数据不会丢失）
